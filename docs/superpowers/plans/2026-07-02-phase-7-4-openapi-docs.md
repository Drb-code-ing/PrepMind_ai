# Phase 7.4 Swagger / OpenAPI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a safe NestJS Swagger / OpenAPI debug entrance for core REST APIs without changing runtime API behavior or weakening auth boundaries.

**Architecture:** Keep `@repo/types` Zod schemas as the primary contract, and add Swagger as a server-side discovery and debugging layer. Use a small config helper to control `/api-docs` and `/api-docs-json`, keep production disabled by default, annotate core controllers with tags/auth/response descriptions, and verify the generated OpenAPI JSON does not contain sensitive examples or raw payload fields.

**Tech Stack:** TypeScript, NestJS 11, `@nestjs/swagger`, Zod, Jest, Bun workspace, PowerShell.

- [ ] **Step 3: Implement helper**

Create `apps/server/src/config/swagger.ts`:

```ts
import type { INestApplication } from '@nestjs/common';
import { DocumentBuilder, SwaggerModule } from '@nestjs/swagger';

import type { ServerEnv } from './env';

export const SWAGGER_UI_PATH = 'api-docs';
export const SWAGGER_JSON_PATH = 'api-docs-json';

type SwaggerEnv = Pick<ServerEnv, 'NODE_ENV' | 'SWAGGER_ENABLED'>;

export function shouldEnableSwagger(env: SwaggerEnv): boolean {
  return env.SWAGGER_ENABLED;
}

export function buildSwaggerDocumentOptions() {
  return new DocumentBuilder()
    .setTitle('PrepMind AI API')
    .setDescription(
      [
        'PrepMind AI server REST API for local development, integration debugging, and interview walkthroughs.',
        'Successful responses are wrapped by the global response envelope: { success: true, data, requestId }.',
        'Error responses are wrapped as: { success: false, error: { code, message }, requestId }.',
        'The source of truth for request and response fields remains @repo/types Zod schemas.',
        'Production Swagger exposure requires an explicit SWAGGER_ENABLED=true override and should only be used in controlled environments.',
      ].join(' '),
    )
    .setVersion('0.7.4')
    .addBearerAuth(
      {
        type: 'http',
        scheme: 'bearer',
        bearerFormat: 'JWT',
        description:
          'Paste a short-lived access token for manual debugging. The web app primary auth flow still uses httpOnly refresh cookies.',
      },
      'access-token',
    )
    .build();
}

export function setupSwagger(
  app: INestApplication,
  env: SwaggerEnv,
): boolean {
  if (!shouldEnableSwagger(env)) {
    return false;
  }

  const document = SwaggerModule.createDocument(
    app,
    buildSwaggerDocumentOptions(),
  );
  SwaggerModule.setup(SWAGGER_UI_PATH, app, document, {
    jsonDocumentUrl: SWAGGER_JSON_PATH,
  });

  return true;
}
```
